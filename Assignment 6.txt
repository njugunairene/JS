class kiosk{
  constructor(fruits){
    this.fruits=fruits || [];    
  }
  addFruit(fruit){
    this.fruits.push(fruit);
  }
  setFruit(fruit){
    this.addFruit(fruit);
  }
  getFruit(){
    return this.fruits;
  }
}
var kiosk = new kiosk(['banana']);
kiosk.setFruit('kiwi')
console.log(kiosk.getFruit());
